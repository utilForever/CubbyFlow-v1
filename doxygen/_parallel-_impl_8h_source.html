28  template <typename RandomIterator, typename RandomIterator2, typename CompareFunction>
29  void Merge(RandomIterator a, size_t size, RandomIterator2 temp, CompareFunction compareFunction)
30  {
31  size_t i1 = 0;
32  size_t i2 = size / 2;
33  size_t tempi = 0;
34 
35  while (i1 < size / 2 && i2 < size)
36  {
37  if (compareFunction(a[i1], a[i2]))
38  {
39  temp[tempi] = a[i1];
40  i1++;
41  }
42  else
43  {
44  temp[tempi] = a[i2];
45  i2++;
46  }
47 
48  tempi++;
49  }
50 
51  while (i1 < size / 2)
52  {
53  temp[tempi] = a[i1];
54  i1++;
55  tempi++;
56  }
57 
58  while (i2 < size)
59  {
60  temp[tempi] = a[i2];

63  }

72  template <typename RandomIterator, typename RandomIterator2, typename CompareFunction>
73  void ParallelMergeSort(RandomIterator a, size_t size, RandomIterator2 temp, unsigned int numThreads, CompareFunction compareFunction)
74  {
75  if (numThreads == 1)
76  {
77  std::sort(a, a + size, compareFunction);
78  }
79  else if (numThreads > 1)
80  {
81  std::vector<std::thread> pool;
82  pool.reserve(2);
83 
84  auto launchRange = [compareFunction](RandomIterator begin, size_t k2, RandomIterator2 temp, unsigned int numThreads)
85  {
86  ParallelMergeSort(begin, k2, temp, numThreads, compareFunction);
87  };
88 
89  pool.emplace_back(launchRange, a, size / 2, temp, numThreads / 2);
90  pool.emplace_back(launchRange, a + size / 2, size - size / 2, temp + size / 2, numThreads - numThreads / 2);
91 
92  // Wait for jobs to finish
93  for (std::thread &t : pool)
94  {
95  if (t.joinable())
96  {
97  t.join();
98  }
99  }
100 
101  Merge(a, size, temp, compareFunction);
102  }

122  // Adopted from http://ideone.com/Z7zldb
123  template <typename IndexType, typename Function>
124  void ParallelFor(IndexType beginIndex, IndexType endIndex, const Function& function)
125  {
126  if (beginIndex > endIndex)
127  {
128  return;
129  }
130 
131  // Estimate number of threads in the pool
132  static const unsigned int numThreadsHint = std::thread::hardware_concurrency();
133  static const unsigned int numThreads = (numThreadsHint == 0u ? 8u : numThreadsHint);
134 
135  // Size of a slice for the range functions
136  IndexType n = endIndex - beginIndex + 1;
137  IndexType slice = static_cast<IndexType>(std::round(n / static_cast<double>(numThreads)));
138  slice = std::max(slice, IndexType(1));
139 
140  // [Helper] Inner loop
141  auto launchRange = [&function](IndexType k1, IndexType k2)
142  {
143  for (IndexType k = k1; k < k2; ++k)
144  {
145  function(k);
146  }
147  };
148 
149  // Create pool and launch jobs
150  std::vector<std::thread> pool;
151  pool.reserve(numThreads);
152  IndexType i1 = beginIndex;
153  IndexType i2 = std::min(beginIndex + slice, endIndex);
154 
155  for (unsigned int i = 0; i + 1 < numThreads && i1 < endIndex; ++i)
156  {
157  pool.emplace_back(launchRange, i1, i2);
158  i1 = i2;
159  i2 = std::min(i2 + slice, endIndex);
160  }
161 
162  if (i1 < endIndex)
163  {
164  pool.emplace_back(launchRange, i1, endIndex);
165  }
166 
167  // Wait for jobs to finish
168  for (std::thread &t : pool)
169  {
170  if (t.joinable())
171  {
172  t.join();
173  }
174  }
175  }
